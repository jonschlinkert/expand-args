## Usage

```js
var expandArgs = require('{%= name %}');
```

### expand args to object values

```js
expandArgs({set: 'a:b'})
//=> {set: {a: 'b'}}

expandArgs({set: 'a.b.c:d'})
//=> {set: {a: {b: {c: 'd'}}}}
```

### expand args to array values

```js
expandArgs({set: 'a:b,c:d'})
//=> {set: [{a: 'b'}, {c: 'd'}]}

expandArgs({set: 'a.b.c:d,e,f'})
//=> {set: {a: {b: {c: ['d', 'e', 'f']}}}}
```

### Convert "object-keys"

```js
expandArgs({'a:b': true})
//=> {a: 'b'}
```

### Expand booleans

```js
expandArgs({set: 'a|b'});
//=> {set: {a: true, b: true}}
```
